using System;

namespace SistemaBecas
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("=== EL RETO: SISTEMA DE BECAS 2.0 (OPTIMIZADO) ===");

            // 1. Pedir los datos al usuario
            Console.Write("Ingrese el promedio obtenido: ");
            double promedio = double.Parse(Console.ReadLine());

            Console.Write("Ingrese la cantidad de faltas: ");
            int faltas = int.Parse(Console.ReadLine());

            // 2. La lógica clave del reto: (Promedio >= 90 Y Faltas <= 2)
            if (promedio >= 90 && faltas <= 2)
            {
                // Mensaje si cumple AMBAS condiciones
                Console.WriteLine("\nFelicidades, aplicas a la beca.");
            }
            else
            {
                // Mensaje si falla alguna de las condiciones
                Console.WriteLine("\nLo sentimos, no cumples con todos los requisitos.");
            }

            // Esperar a que el usuario presione una tecla para cerrar
            Console.WriteLine("\nPresione cualquier tecla para salir...");
            Console.ReadKey();
        }
    }
}
